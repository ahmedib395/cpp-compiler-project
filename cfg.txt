# Context Free Grammar (CFG) for Phase 1 - 4
# Top-Down Parser (Recursive Descent Expansion)
# Uses exact terminal token types matched in lexer.py.

Program -> IncludeStmt UsingStmt GlobalList

# Mandatory Boilerplate
IncludeStmt -> PREPROCESSOR
UsingStmt -> USING NAMESPACE IDENTIFIER SEMI

# List of global entities (EBNF repetition for Top-Down parsing)
GlobalList -> { Global }
Global -> FunctionDef | Declaration | Statement

FunctionDef -> Type MAIN LPAREN ParamList RPAREN LBRACE StatementList RBRACE
             | Type IDENTIFIER LPAREN ParamList RPAREN LBRACE StatementList RBRACE

ParamList -> Param { COMMA Param } | empty
Param -> Type IDENTIFIER

Type -> INT | FLOAT | DOUBLE | CHAR | BOOL | VOID | CONST INT | CONST FLOAT | CONST DOUBLE

StatementList -> { Statement }
Statement -> Declaration \
           | Assignment \
           | WhileLoop \
           | ForLoop \
           | DoWhile \
           | IfStatement \
           | BreakStmt \
           | ContinueStmt \
           | CinStmt \
           | CoutStmt \
           | ReturnStmt \
           | Expression SEMI

Declaration -> Type IDENTIFIER SEMI | Type IDENTIFIER ASSIGN Expression SEMI | Type VarList SEMI
VarList -> VarItem { COMMA VarItem }
VarItem -> IDENTIFIER | IDENTIFIER ASSIGN Expression

Assignment -> IDENTIFIER ASSIGN Expression SEMI \
            | IDENTIFIER PLUS_ASSIGN Expression SEMI \
            | IDENTIFIER MINUS_ASSIGN Expression SEMI \
            | IDENTIFIER TIMES_ASSIGN Expression SEMI \
            | IDENTIFIER DIV_ASSIGN Expression SEMI \
            | IDENTIFIER INC SEMI \
            | IDENTIFIER DEC SEMI

WhileLoop -> WHILE LPAREN Condition RPAREN BlockOrStmt
ForLoop -> FOR LPAREN ForInit Condition SEMI ForUpdate RPAREN BlockOrStmt
ForInit -> Type IDENTIFIER ASSIGN Expression SEMI | IDENTIFIER ASSIGN Expression SEMI | SEMI
ForUpdate -> IDENTIFIER INC | IDENTIFIER DEC | IDENTIFIER ASSIGN Expression | IDENTIFIER PLUS_ASSIGN Expression | IDENTIFIER MINUS_ASSIGN Expression | empty
DoWhile -> DO BlockOrStmt WHILE LPAREN Condition RPAREN SEMI

IfStatement -> IF LPAREN Condition RPAREN BlockOrStmt ElsePart
ElsePart -> ELSE IF LPAREN Condition RPAREN BlockOrStmt ElsePart \
          | ELSE BlockOrStmt \
          | empty

BlockOrStmt -> LBRACE StatementList RBRACE | Statement

BreakStmt -> BREAK SEMI
ContinueStmt -> CONTINUE SEMI
ReturnStmt -> RETURN Expression SEMI | RETURN SEMI

CinStmt -> CIN { SHR IDENTIFIER } SEMI
CoutStmt -> COUT { SHL CoutAtom } SEMI
CoutAtom -> Expression | STRING | ENDL

Condition -> AndCond { OR AndCond }
AndCond -> CondAtom { AND CondAtom }
CondAtom -> NOT CondAtom | Expression RelOp Expression | Expression

RelOp -> EQ | NEQ | LT | GT | LEQ | GEQ

# Math expressions (Top-Down precedence levels)
Expression -> Term { (PLUS | MINUS) Term }
Term -> Unary { (TIMES | DIVIDE | MOD) Unary }
Unary -> MINUS Factor | NOT Factor | Factor

Factor -> LPAREN Expression RPAREN \
        | IDENTIFIER LPAREN ArgList RPAREN \
        | IDENTIFIER \
        | NUMBER \
        | FLOAT_NUM \
        | STRING \
        | CHAR_LIT \
        | TRUE \
        | FALSE

ArgList -> Expression { COMMA Expression } | empty
